1. TABLAS PRINCIPALES
Usuarios
CREATE TABLE usuarios (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    email VARCHAR(100) UNIQUE NOT NULL,
    contraseña VARCHAR(255) NOT NULL,
    rol ENUM('admin', 'estudiante') DEFAULT 'estudiante',
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    activo BOOLEAN DEFAULT TRUE,
    ultimo_acceso DATETIME
);

Datos específicos de estudiantes
CREATE TABLE estudiantes (
    id_estudiante INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT UNIQUE NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    apellidos VARCHAR(100),
    fecha_nacimiento DATE,
    telefono VARCHAR(20),
    direccion TEXT,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE
);

Datos específicos de administradores
CREATE TABLE administradores (
    id_admin INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT UNIQUE NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    permisos JSON COMMENT 'Permisos adicionales del admin',
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE
);

MÓDULO DE EXÁMENES
-- Categorías de preguntas
CREATE TABLE categorias_pregunta (
    id_categoria INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50) NOT NULL,
    peso DECIMAL(3,2) DEFAULT 1.00 COMMENT 'Peso en la calificación',
    descripcion TEXT
);

Exámenes
CREATE TABLE examenes (
    id_examen INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(200) NOT NULL,
    descripcion TEXT,
    nivel ENUM('basico', 'intermedio', 'avanzado') DEFAULT 'basico',
    duracion_minutos INT DEFAULT 60,
    fecha_inicio DATETIME,
    fecha_cierre DATETIME,
    estado ENUM('borrador', 'activo', 'pausado', 'finalizado') DEFAULT 'borrador',
    puntaje_maximo INT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME ON UPDATE CURRENT_TIMESTAMP,
    creado_por INT,
    FOREIGN KEY (creado_por) REFERENCES administradores(id_admin)
);

Preguntas
CREATE TABLE preguntas (
    id_pregunta INT PRIMARY KEY AUTO_INCREMENT,
    id_examen INT NOT NULL,
    id_categoria INT,
    texto TEXT NOT NULL,
    tipo ENUM('opcion_multiple', 'verdadero_falso', 'texto') DEFAULT 'opcion_multiple',
    opcion_a VARCHAR(255),
    opcion_b VARCHAR(255),
    opcion_c VARCHAR(255),
    opcion_d VARCHAR(255),
    respuesta_correcta CHAR(1) CHECK (respuesta_correcta IN ('a', 'b', 'c', 'd', NULL)),
    puntos INT DEFAULT 1,
    orden INT DEFAULT 0,
    FOREIGN KEY (id_examen) REFERENCES examenes(id_examen) ON DELETE CASCADE,
    FOREIGN KEY (id_categoria) REFERENCES categorias_pregunta(id_categoria)
);

 ACCESO Y SEGUIMIENTO
-- Códigos de acceso (para estudiantes)
CREATE TABLE codigos_acceso (
    id_codigo INT PRIMARY KEY AUTO_INCREMENT,
    codigo VARCHAR(20) UNIQUE NOT NULL,
    id_examen INT NOT NULL,
    usos_max INT DEFAULT 1,
    usos_actuales INT DEFAULT 0,
    fecha_vencimiento DATETIME,
    estado ENUM('disponible', 'usado', 'expirado', 'revocado') DEFAULT 'disponible',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    creado_por INT,
    FOREIGN KEY (id_examen) REFERENCES examenes(id_examen),
    FOREIGN KEY (creado_por) REFERENCES administradores(id_admin)
);

Asignación estudiante-código
CREATE TABLE asignacion_codigo (
    id_asignacion INT PRIMARY KEY AUTO_INCREMENT,
    id_estudiante INT NOT NULL,
    id_codigo INT NOT NULL,
    fecha_asignacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_estudiante) REFERENCES estudiantes(id_estudiante),
    FOREIGN KEY (id_codigo) REFERENCES codigos_acceso(id_codigo),
    UNIQUE KEY unique_asignacion (id_estudiante, id_codigo)
);

Sesiones de examen
CREATE TABLE sesiones_examen (
    id_sesion INT PRIMARY KEY AUTO_INCREMENT,
    id_estudiante INT NOT NULL,
    id_examen INT NOT NULL,
    id_codigo INT NOT NULL,
    fecha_inicio DATETIME DEFAULT CURRENT_TIMESTAMP,
    fecha_fin DATETIME,
    tiempo_transcurrido INT COMMENT 'Segundos totales',
    estado ENUM('en_curso', 'pausado', 'finalizado', 'abandonado') DEFAULT 'en_curso',
    ip_address VARCHAR(45),
    user_agent TEXT,
    FOREIGN KEY (id_estudiante) REFERENCES estudiantes(id_estudiante),
    FOREIGN KEY (id_examen) REFERENCES examenes(id_examen),
    FOREIGN KEY (id_codigo) REFERENCES codigos_acceso(id_codigo)
);

RESULTADOS Y EVALUACIÓN
-- Resultados de examen
CREATE TABLE resultados_examen (
    id_resultado INT PRIMARY KEY AUTO_INCREMENT,
    id_sesion INT NOT NULL,
    puntaje_total INT,
    puntaje_obtenido INT,
    porcentaje DECIMAL(5,2),
    estado ENUM('aprobado', 'reprobado', 'pendiente_revision', 'anulado') DEFAULT 'pendiente_revision',
    fecha_calificacion DATETIME,
    revisado_por INT,
    observaciones TEXT,
    FOREIGN KEY (id_sesion) REFERENCES sesiones_examen(id_sesion),
    FOREIGN KEY (revisado_por) REFERENCES administradores(id_admin)
);

Respuestas del estudiante (por pregunta)
CREATE TABLE respuestas_estudiante (
    id_respuesta INT PRIMARY KEY AUTO_INCREMENT,
    id_sesion INT NOT NULL,
    id_pregunta INT NOT NULL,
    respuesta_seleccionada CHAR(1),
    respuesta_texto TEXT COMMENT 'Para preguntas tipo texto',
    es_correcta BOOLEAN DEFAULT FALSE,
    puntaje_obtenido INT DEFAULT 0,
    tiempo_segundos INT COMMENT 'Tiempo en esta pregunta',
    revisado BOOLEAN DEFAULT FALSE,
    observacion TEXT,
    FOREIGN KEY (id_sesion) REFERENCES sesiones_examen(id_sesion),
    FOREIGN KEY (id_pregunta) REFERENCES preguntas(id_pregunta)
);

AUDITORÍA Y SEGUIMIENTO
Temporizador por pregunta
CREATE TABLE tiempos_pregunta (
    id_tiempo INT PRIMARY KEY AUTO_INCREMENT,
    id_respuesta INT NOT NULL,
    tiempo_inicio DATETIME,
    tiempo_fin DATETIME,
    duracion_segundos INT,
    FOREIGN KEY (id_respuesta) REFERENCES respuestas_estudiante(id_respuesta)
);

Notificaciones del sistema
CREATE TABLE notificaciones (
    id_notificacion INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT NOT NULL,
    tipo ENUM('info', 'warning', 'success', 'danger') DEFAULT 'info',
    titulo VARCHAR(100),
    mensaje TEXT,
    leida BOOLEAN DEFAULT FALSE,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    fecha_lectura DATETIME,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
);

 Historial de cambios
CREATE TABLE historial_cambios (
    id_cambio INT PRIMARY KEY AUTO_INCREMENT,
    tabla_afectada VARCHAR(50),
    id_registro INT,
    accion ENUM('INSERT', 'UPDATE', 'DELETE'),
    datos_previos JSON,
    datos_nuevos JSON,
    fecha_cambio DATETIME DEFAULT CURRENT_TIMESTAMP,
    usuario_id INT,
    ip_address VARCHAR(45),
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id_usuario)
);
